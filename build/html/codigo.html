 1-- Criação das Triggers
 2CREATE OR REPLACE FUNCTION matriz.fn_sala_auditoria()
 3RETURNS trigger AS
 4$$
 5       BEGIN
 6      IF(tg_op = 'DELETE') THEN
 7         INSERT INTO tb_sala_auditoria
 8         SELECT 'E', user, now(),OLD.*;
 9         RETURN OLD;
10      ELSIF(tg_op = 'UPDATE') THEN
11         INSERT INTO tb_sala_auditoria
12         SELECT 'A', user, now(),OLD.*;
13         INSERT INTO tb_sala_auditoria
14         SELECT 'A', user, now(),NEW.*;
15         RETURN NEW;
16      ELSIF(tg_op = 'INSERT') THEN
17         INSERT INTO tb_sala_auditoria
18         SELECT 'I', user, now(),NEW.*;
19         RETURN NEW;
20      END IF;
21      RETURN NULL;
22   END
23$$
24LANGUAGE plpgsql;

1create trigger tg_sala_auditoria
2after insert or update or delete on matriz.tb_sala
3for each row execute procedure matriz.fn_sala_auditoria();
