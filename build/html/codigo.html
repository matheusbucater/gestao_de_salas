 86-- Criação das Triggers
 87CREATE OR REPLACE FUNCTION matriz.fn_sala_auditoria()
 88RETURNS trigger AS
 89$$
 90	BEGIN
 91    	IF(tg_op = 'DELETE') THEN
 92           	INSERT INTO tb_sala_auditoria
 93            SELECT 'E', user, now(),OLD.*;
 94            RETURN OLD;
 95    	ELSIF(tg_op = 'UPDATE') THEN
 96           	INSERT INTO tb_sala_auditoria
 97            SELECT 'A', user, now(),OLD.*;
 98           	INSERT INTO tb_sala_auditoria
 99            SELECT 'A', user, now(),NEW.*;
100            RETURN NEW;
101    	ELSIF(tg_op = 'INSERT') THEN
102           	INSERT INTO tb_sala_auditoria
103            SELECT 'I', user, now(),NEW.*;
104            RETURN NEW;
105        END IF;
106        RETURN NULL;                   
107    END
108$$
109LANGUAGE plpgsql;

111create trigger tg_sala_auditoria
112after insert or update or delete on matriz.tb_sala 
113for each row execute procedure matriz.fn_sala_auditoria();
